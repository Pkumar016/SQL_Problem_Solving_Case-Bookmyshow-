// Query Retrive

SELECT s.Screen_Number, s.Show_Time, m.Movie_Name
FROM Show s
JOIN Movie m ON s.Movie_ID = m.Movie_ID
JOIN Date d ON s.Date_ID = d.Date_ID
JOIN Theatre t ON s.Theatre_ID = t.Theatre_ID
WHERE d.Date = '2024-03-27'  -- Specify the date
AND t.Theatre_Name = 'Example Theatre'; -- Specify the theatre name
